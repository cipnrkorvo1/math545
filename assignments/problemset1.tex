\begin{align*}
	\begin{augmx}{3}
		-1 & 1 & 4 & -1\\
		3 & -1 & 2 & 2\\
		2 & -2 & -8 & 2
	\end{augmx}&\\
	{R_3 + 2R_1 \rightarrow R_3}\quad \xlrightarrow
	\begin{augmx}{3}
		-1 & 1 & 4 & -1\\
		3 & -1 & 2 & 2\\
		0 & 0 & 0 & 0
	\end{augmx}&\\
	{R_2 + 3R_1 \rightarrow R_2}\quad \xlrightarrow
	\begin{augmx}{3}
		-1 & 1 & 4 & -1\\
		0 & 2 & 14 & -1\\
		0 & 0 & 0 & 0
	\end{augmx}&\\
	\\
	\xlrightarrow
	\begin{linsys}{3}
		x &- &y &- &4z &= &1\\
		& &2y &+ &14z &= &-1\\
		& & & &0z &= &0
	\end{linsys}\\
\end{align*}
Perform back substitution:
\vspace{-1.5em}
\begin{multicols}{3}
	\begin{align*}
		z = t\\
		\text{($z$ is a}\\ \text{parameter/free variable)}
	\end{align*}
	\columnbreak\\
	\begin{align*}
		2y + 14z &= -1\\
		2y &= -1 - 14t\\
		y &= -7t - \frac{1}{2}
	\end{align*}
	\columnbreak\\
	\begin{align*}
		x - y - 4z &= 1\\
		x &= y + 4t + 1\\
		x &= (-7t - \frac{1}{2}) + 4t + 1\\
		x &= -3t + \frac{1}{2}
	\end{align*}
\end{multicols}
\[
	\boxed{S = \left\{(-3t + \frac{1}{2}, -7t - \frac{1}{2}, t) \; | \; t \in \reals\right\}}
\]

Perform back substitution:
\vspace{-1em}
\begin{multicols}{2}
	\begin{align*}
		y& + 2z = -a - 3b\\
		y& = -2z - a - 3b\\
		y& = -2(b-c) - a - 3b\\

	\end{align*}
	\columnbreak\\
	\begin{align*}
		x& - y - z = b\\
		x& = y + z + b\\
		x& = (-a - 5b + 2c) + (b - c) + b\\
		x& = -a -3b + c
	\end{align*}
\end{multicols}
