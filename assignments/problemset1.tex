	\end{augmx}&\\

	\begin{augmx}{3}

		0 & 0 & 0 & 0
	\end{augmx}&\\
	\\
	\xlrightarrow
	\begin{linsys}{3}
		x &- &y &- &4z &= &1\\
		& &2y &+ &14z &= &-1\\
		& & & &0z &= &0
	\end{linsys}\\
\end{align*}
Perform back substitution:
\vspace{-1.5em}
\begin{multicols}{3}
	\begin{align*}
		z = t\\
		\text{($z$ is a}\\ \text{parameter/free variable)}
	\end{align*}
	\columnbreak\\
	\begin{align*}
		2y + 14z &= -1\\
		2y &= -1 - 14t\\
		y &= -7t - \frac{1}{2}
	\end{align*}
	\columnbreak\\
	\begin{align*}
		x - y - 4z &= 1\\
		x &= y + 4t + 1\\
		x &= (-7t - \frac{1}{2}) + 4t + 1\\
		x &= -3t + \frac{1}{2}
	\end{align*}
\end{multicols}
\[
	\boxed{S = \left\{(-3t + \frac{1}{2}, -7t - \frac{1}{2}, t) \; | \; t \in \reals\right\}}
\]
