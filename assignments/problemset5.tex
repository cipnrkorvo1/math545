\section{}
Let $S_1$ be the set of all linear combinations of the vectors $\vb{v}_1, \vb{v}_2, ..., \vb{v}_k$ in $\reals^n$.\\\noindent
Let $S_2$ be the set of all linear combinations of the vectors $\vb{v}_1, \vb{v}_2, ..., \vb{v}_k, \vb{v}_1 + \vb{v}_2$.\\\noindent
Show that $S_1 = S_2$ by showing that each of the sets is a subset of the other.
\begin{proof}$\;$\\
	(1)  \begin{adjustwidth}{0.5cm}{}
	The definition of $S_2$ indicates $S_2 = S_1 \cup \vb{v}_1 + \vb{v_2}, \implies S_1 \subseteq S_2$.\\
	\end{adjustwidth}
	(2) \begin{adjustwidth}{0.5cm}{}
	By the definition of linear combinations,
	\[ S_1 = c_1\vb{v}_1 + c_2\vb{v}_2 + \cdots + c_k\vb{v}_k = \sum_{i=1}^{k}c_i\vb{v}_i \]
	Using $c_1 = 1, c_2 = 1,$ and $c_i = 0$ for all $i > 2$, 
	\[ 1\vb{v}_1 + 1\vb{v}_2 + 0\vb{v}_3 + \cdots + 0\vb{v}_k = \vb{v}_1 + \vb{v}_2 + 0[\vb{v}_3 + \cdots + \vb{v}_k] = \vb{v}_1 + \vb{v}_2 \]
	is necessarily in $S_1$.\\
	$\therefore \vb{v}_1 + \vb{v}_2 \in S_1$ which implies $S_1 \cup \vb{v}_1 + \vb{v}_2 \subseteq S_1$, therefore $S_2 \subseteq S_1$ by definition of $S_2$.
	\end{adjustwidth}
	Since we have shown $S_1 \subseteq S_2$ and $S_2 \subseteq S_1$, $S_1 = S_2$.
\end{proof}
\section{}
Suppose that $S = \{ \vb{v}_1, \vb{v}_2, \vb{v}_3 \}$ is linearly independent and further suppose that
\[ \vb{w}_1 \vb{v}_2\;, \vb{w}_2 = \vb{v}_1 + \vb{v}_3\;\text{, and } \vb{w}_3 = \vb{v}_1 + \vb{v}_2 + \vb{v}_3\]
Determine whether the set $T = \{\vb{w}_1, \vb{w}_2, \vb{w}_3\}$ is linearly dependent or linearly independent.\\\\

\textbf{Claim:} $T$ is linearly dependent.
\begin{proof}
	We will prove the claim by showing that there is a nontrivial solution to the equation
	\[ c_1\vb{w}_1 + c_2\vb{w}_2 + c_3\vb{w}_3 = 0 \]
	i.e. a solution where at least one $c_1, c_2, c_3 \neq 0$. First, substitute all $\vb{w}_i$ with the corresponding sum of vectors $\vb{v}$:
	\[ c_1\vb{v}_2 + c_2[\vb{v}_1 + \vb{v}_3] + c_3[\vb{v}_1 + \vb{v}_2 + \vb{v}_3] = 0 \]
	Let $c_3 \neq 0$ and $c_1 = c_2 = -c_3$, $\; \implies c_1,c_2,c_3 \neq 0$.
	\[ -c_3\vb{v}_2 - c_3[\vb{v}_1 + \vb{v}_3] + c_3[\vb{v}_1 + \vb{v}_2 + \vb{v}_3] = 0\]
	Reducing this equation, we get
	\[ -c_3[\vb{v}_1 + \vb{v}_2 + \vb{v}_3] + c_3[\vb{v}_1 + \vb{v}_2 + \vb{v}_3] = 0 \longrightarrow 0 = 0\]
	Thus we have found a nontrivial solution to the equation.\\
	$\therefore$ $T$ is linearly dependent.
\end{proof}
