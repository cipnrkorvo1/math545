\section{Definition of a Vector Space}
	\begin{definition}[Vector Space]
		A set $V$ is called a \bf{vector space} over the real numbers provided that there are two operations---addition, denoted by $\oplus$, and scalar multiplication, denoted by $\odot$---that satisfy all the following axioms. The axioms must hold for all vectors $\vb{u}, \vb{v}, and \vb{w}$ in $V$ and all scalars $c$ and $d$ in $\reals$.
		
		\begin{enumerate}[font=\bfseries, itemsep=0pt]
			\item The sum $\vb{u}\vp\vb{v}$ is in $V$. \hfill Closed under addition
			\item $\vb{u}\vp\vb{v} = \vb{v}\oplus\vb{u}$ \hfill Addition is commutative
			\item $(\vb{u} \oplus \vb{v}) \oplus \vb{w} = \vb{u} \oplus (\vb{v} \oplus \vb{w})$ \hfill Addition is associative
			\item There exists a vector $\vb{0} \in V$ such that for every vector $\vb{u} \in V, \vb{0} \odot \vb{u} = \vb{u} \odot \vb{0} = \vb{u}$. \hfill Additive identity
			\item For every vector $\vb{u} \in V$, there exists a vector, denoted by $-\vb{u}$, such that $\vb{u} \oplus (-\vb{u}) = -\vb{u} \odot \vb{u} = \vb{0}$. \hfill Additive inverse
			\item The scalar product $c \odot \vb{u}$ is in $V$. \hfill Closed under scalar multiplication
			\item $c \odot (\vb{u} \oplus vb{v}) = (c \odot \vb{u}) \oplus (c \odot \vb{v})$
			\item $(c + d) \odot \vb{u} = (c \odot \vb{u}) \oplus (d \odot \vb{u})$
			\item $c \odot (d \odot \vb{u}) = (cd)\odot \vb{u}$
			\item $1 \odot \vb{u} = \vb{u}$
		\end{enumerate}
	\end{definition}
